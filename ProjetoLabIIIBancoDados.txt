CREATE DATABASE "ProjetoLabIII"
  WITH OWNER = postgres
       ENCODING = 'UTF8'
       TABLESPACE = pg_default
       LC_COLLATE = 'Portuguese_Brazil.1252'
       LC_CTYPE = 'Portuguese_Brazil.1252'
       CONNECTION LIMIT = -1;

create table Alunos
(
NOME varchar(60),
IDADE NUMERIC NOT NULL,
Endereco varchar(60));


create table Treinos
(
Exercícios varchar(60),
Series numeric,
Repeticoes numeric);
